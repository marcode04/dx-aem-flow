<SectionHeading

/>
<Section>
  <div class="grid grid-cols-1 md:grid-cols-3 gap-6">
    <ContentCard icon="mdi:pin" iconColor="bg-amber-500" title="/aem-init">
      Run once per repo. Builds <code>.ai/project/</code> with component-index, file-patterns,
      content-paths, and architecture docs.
    </ContentCard>

    <ContentCard icon="mdi:sync" iconColor="bg-cyan" title="/aem-refresh">
      Syncs seed data when project structure changes. Updates component index and file patterns.
    </ContentCard>

    <ContentCard icon="mdi:magnify" iconColor="bg-brand-700" title="/aem-component">
      Finds all source files for a named component: JS, SCSS, HTL, dialog XML, Sling Model,
      content paths.
    </ContentCard>

    <ContentCard icon="mdi:file-search" iconColor="bg-emerald-600" title="/aem-page-search">
      Find all AEM pages that use a specific component. Returns page paths with clickable
      author URLs grouped by site and language.
    </ContentCard>

    <ContentCard icon="mdi:hospital-box" iconColor="bg-red-500" title="/aem-doctor">
      Health check for AEM infrastructure -- verifies component definitions, OSGi configs,
      dispatcher rules, content structure, and AEM instance connectivity.
    </ContentCard>
  </div>

  <HighlightBox severity="info" title="Seed Data Files">
    All stored in <code>.ai/project/</code>: <code>project.yaml</code>, <code>file-patterns.yaml</code>,
    <code>content-paths.yaml</code>, <code>component-index-project.md</code>, <code>architecture.md</code>,
    <code>features.md</code>.
  </HighlightBox>
</Section>
